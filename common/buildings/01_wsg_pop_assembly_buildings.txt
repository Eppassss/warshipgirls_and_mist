#港区建造器
building_warshipgirl_assembly_plant = {
	base_buildtime = 240
	base_cap_amount = 1

	category = pop_assembly
	
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
			has_authority = auth_warshipgirls	
			NOT = { has_ascension_perk = ap_project_s }
		}
	}
	
	convert_to = {
		building_warshipgirl_senergy_assembly_plant
	}
	
	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				OR = {
					is_regular_empire = no 
				}
			}
			has_modifier = slave_colony
			AND = {
				owner = { is_ai = yes }
				free_jobs = 0
				OR = {
					num_buildings = {
						type = any
						value = 16
					}
					is_planet_class = pc_habitat
				}
			}
			owner = {
				has_ascension_perk = ap_project_s
			}
		}
	}
	
	planet_modifier = {
		planet_pop_assembly_add = 3
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			minerals = 5
			energy = 5
		}
	}	

	prerequisites = {
		"tech_warshipgirl_assembly_plant"
	}
	
	upgrades = {
		building_warshipgirl_assembly_plantU
	}
		
	ai_weight = {
		weight = 100
	}
}
